%% ===============[ Reserved ]=============== %%

% Set the style of reserved words
\reservestyle[\mathinner]{\command}{\mathbf}

\newcommand{\McmdItem}[1]{\command{#1[\;#1\;]}}

\newcommand{\lcmd}{\forcsvlist{\LcmdItem}}
\newcommand{\mcmd}{\forcsvlist{\McmdItem}}
\newcommand{\rcmd}{\forcsvlist{\RcmdItem}}
